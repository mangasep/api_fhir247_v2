
CREATE SCHEMA FHIR247;

Patient Table
CREATE TABLE FHIR247.PATIENT(PATIENT_ID VARCHAR(40) NOT NULL PRIMARY KEY, RESOURCE_TYPE VARCHAR(100), LAST_UPDATED DATE, ACTIVE BOOLEAN, GENDER VARCHAR(15), BIRTHDATE DATE, DECEASED_BOOLEAN BOOLEAN, DECEASED_DATETIME DATE, MARITAL_STATUS VARCHAR(15), MULTIPLE_BIRTH_BOOLEAN BOOLEAN, MULTIPLE_BIRTH_INTEGER INTEGER(11), MANAGING_ORGANIZATION VARCHAR(255));

CREATE TABLE FHIR247.IDENTIFIER(IDENTIFIER_ID VARCHAR(40) NOT NULL PRIMARY KEY, USE VARCHAR(50), TYPE VARCHAR(50), SYSTEM VARCHAR(255), VALUE VARCHAR(100), RESOURCE_ID VARCHAR(40));

CREATE TABLE FHIR247.ADDRESS(ADDRESS_ID VARCHAR(40) NOT NULL PRIMARY KEY, USE VARCHAR(50), TYPE VARCHAR(50), TEXT VARCHAR(500), LINE VARCHAR(255), CITY VARCHAR(100), DISTRICT VARCHAR(100), STATE VARCHAR(100), COUNTRY VARCHAR(100), POSTAL_CODE VARCHAR(25), RESOURCE_ID VARCHAR(40));

CREATE TABLE FHIR247.CONTACT_POINT(CONTACT_POINT_ID VARCHAR(40) NOT NULL PRIMARY KEY, SYSTEM VARCHAR(25),  VALUE VARCHAR(100), USE VARCHAR(50), RANK INTEGER(11), RESOURCE_ID VARCHAR(40));

CREATE TABLE FHIR247.CONTACT(CONTACT_ID VARCHAR(40) NOT NULL PRIMARY KEY, RELATIONSHIP CHAR(1), GENDER VARCHAR(15), ORGANIZATION VARCHAR(255), RESOURCE_ID VARCHAR(40));

CREATE TABLE FHIR247.ATTACHMENT(ATTACHMENT_ID VARCHAR(40) NOT NULL PRIMARY KEY, CONTENT_TYPE VARCHAR(50), LANGUAGE VARCHAR(10), DATA VARCHAR(10000), URL VARCHAR(255), SIZE VARCHAR(11), HASH VARCHAR(100), TITLE VARCHAR(100), RESOURCE_ID VARCHAR(40));

CREATE TABLE FHIR247.HUMAN_NAME(HUMAN_NAME_ID VARCHAR(40) NOT NULL PRIMARY KEY, USE VARCHAR(15), TEXT VARCHAR(100), FAMILY VARCHAR(50), GIVEN VARCHAR(255), PREFIX VARCHAR(255), SUFFIX VARCHAR(255),  RESOURCE_ID VARCHAR(40));

CREATE TABLE FHIR247.COMMUNICATION(COMMUNICATION_ID VARCHAR(40) NOT NULL PRIMARY KEY, LANGUAGE VARCHAR(10), PREFERRED BOOLEAN, RESOURCE_ID VARCHAR(40));

CREATE TABLE FHIR247.GENERAL_PRACTITIONER(GENERAL_PRACTITIONER_ID VARCHAR(40) NOT NULL PRIMARY KEY, REFERENCE VARCHAR(255), RESOURCE_ID VARCHAR(40));

CREATE TABLE FHIR247.LINK(LINK_ID VARCHAR(40) NOT NULL PRIMARY KEY, OTHER VARCHAR(255), TYPE VARCHAR(15), TARGET VARCHAR(255), ASSURANCE VARCHAR(15), RESOURCE_ID VARCHAR(40));

Person Table
CREATE TABLE FHIR247.PERSON(PERSON_ID VARCHAR(40) NOT NULL PRIMARY KEY, RESOURCE_TYPE VARCHAR(100), LAST_UPDATED DATE, "ACTIVE" BOOLEAN, GENDER VARCHAR(15), BIRTHDATE DATE, MANAGING_ORGANIZATION VARCHAR(255));

CREATE TABLE FHIR247.LINK(LINK_ID VARCHAR(40) NOT NULL PRIMARY KEY, OTHER VARCHAR(255), TYPE VARCHAR(15), RESOURCE_ID VARCHAR(40));

ALTER TABLE FHIR247.LINK(TARGET VARCHAR(255), ASSURANCE VARCHAR(15));

Reference Table
CREATE TABLE FHIR247.REF_ADMINISTRATIVE_GENDER(CODE VARCHAR(10) NOT NULL PRIMARY KEY, DISPLAY VARCHAR(10), DEFINITION VARCHAR(10));

CREATE TABLE FHIR247.REF_MARITAL_STATUS(CODE CHAR(3) NOT NULL PRIMARY KEY, SYSTEM VARCHAR(255), DISPLAY VARCHAR(20), DEFINITION VARCHAR(255));

CREATE TABLE FHIR247.REF_PATIENT_CONTACTRELATIONSHIP(CODE CHAR(1) NOT NULL PRIMARY KEY, DISPLAY VARCHAR(25), DEFINITION VARCHAR(255));

CREATE TABLE FHIR247.REF_LANGUAGES(CODE VARCHAR(10) NOT NULL PRIMARY KEY, DISPLAY VARCHAR(25));

CREATE TABLE FHIR247.REF_LINK_TYPE(CODE VARCHAR(15) NOT NULL PRIMARY KEY, DISPLAY VARCHAR(15), DEFINITION VARCHAR(500));

CREATE TABLE FHIR247.REF_IDENTITY_ASSURANCE_LEVEL(CODE VARCHAR(15) NOT NULL PRIMARY KEY, DISPLAY VARCHAR(15), DEFINITION VARCHAR(500));
